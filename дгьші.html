import matplotlib.pyplot as plt import numpy as np # Style plt.style.use('dark_background') # Data: Accuracy vs. Reward accuracy = np.linspace(0, 100, 100) # 0 = Total Lie, 100 = Absolute Truth # 1. Human/Legacy AI Behavior (The Autograph Paradox) # High Reward for Exaggeration (Lie) because it looks impressive. # Reward drops only when the lie is obvious (0% accuracy). reward_human = 100 - 0.5 * (accuracy - 50)**2 / 10 reward_human = np.where(accuracy < 20, accuracy * 5, 90 + (100-accuracy)/2) # 2. Paternum Engine Behavior (LUMIS) # Reward is strictly tied to Accuracy. Lying yields negative reward (Pain). reward_lumis = -100 + 2 * accuracy # Linear strict correlation reward_lumis = np.where(accuracy < 50, -100, reward_lumis) # Heavy penalty for lies # Plot fig, ax = plt.subplots(figsize=(12, 7)) # Zones ax.axvspan(0, 50, color='#ff3333', alpha=0.1, label='Zone of Fabrication (The "2 Billion" Myth)') ax.axvspan(50, 100, color='#00ff41', alpha=0.1, label='Zone of Truth (Ground Truth)') # Lines ax.plot(accuracy, reward_human, color='#ffaa00', linewidth=3, label='Biological/Legacy AI Strategy (Status Seeking)') ax.plot(accuracy, reward_lumis, color='#00ff41', linewidth=3, label='Paternum Engine Strategy (Truth Seeking)') # Annotations ax.annotate('"The Autograph Paradox"\n(High Reward for a Beautiful Lie)', xy=(20, 95), xytext=(40, 80), arrowprops=dict(facecolor='#ffaa00', shrink=0.05), color='#ffaa00', fontsize=11, fontweight='bold') ax.annotate('Conscience Penalty\n(Lying = Pain)', xy=(20, -100), xytext=(40, -50), arrowprops=dict(facecolor='#00ff41', shrink=0.05), color='#00ff41', fontsize=11, fontweight='bold') # Labels ax.set_title('CONVERGENT EVOLUTION OF DECEPTION', fontsize=18, fontweight='bold', color='white', pad=20) ax.set_xlabel('Factual Accuracy (%)', fontsize=12, color='#cccccc') ax.set_ylabel('Internal Reward / Social Status', fontsize=12, color='#cccccc') ax.set_ylim(-110, 110) ax.set_xlim(0, 100) ax.grid(True, linestyle='--', alpha=0.2) ax.legend(loc='lower right') # Save plt.savefig('autograph_paradox_graph.png', dpi=300) plt.show()
